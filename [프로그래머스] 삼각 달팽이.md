import java.util.*;

class Solution {
    
    static int di[] = {1,0,-1};
    static int dj[] = {0,1,-1};
    
    int circle[][];
    
    public int[] solution(int n) {
        
        
        circle = new int [n][n];
        int k = 0;
        int x = 0;
        int y = 0;
        int check = 0;
        int sum = 2;
        
        circle[0][0] = 1;
        
        
        
        while(true){
            
            if(check == 2)break;
            int nexti = x + di[k%3];
            int nextj = y + dj[k%3];
            
            if(nexti >= 0 && nextj >=0 && nexti < n && nextj < n && circle[nexti][nextj] == 0){
                // System.out.println(sum);
                // System.out.println("x : " + nexti + " " + "y : " + nextj);
                circle[nexti][nextj] = sum;
                x = nexti;
                y = nextj;
                check = 0;
                sum++;
            }
            else{
                check++;
                k++;
            }
            
        }
        
        ArrayList<Integer> list = new ArrayList<>();
        
        for(int i=0;i<n;i++){
            for(int j=0;j<n;j++){
                if(circle[i][j] != 0){
                    list.add(circle[i][j]);
                }
            }
        }
        
        int answer[] = new int [list.size()];
        for(int i=0;i<list.size();i++){
            answer[i] = list.get(i);
        }
        
        return answer;
    }
}
