import java.util.*;
class Solution {
    public int[] solution(int n, int s) {
                
        if(n > s){
            int answer[] = {-1};
            
            return answer;
        }
        
        int size = n;
               
        int[] answer = new int [n];
        
        for(int i=0;i<size;i++){
            int div = s/n;
            answer[i] = div; 
            s -= div;
            n--;
            
        }

        Arrays.sort(answer);
        
        return answer;
    }
}
