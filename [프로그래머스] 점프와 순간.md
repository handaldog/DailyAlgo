public class Solution {
    public int solution(int n) {
        
         int ans = 0;
        
        int result[] = {0,1,1,2,1};
        
        while(n > 4){
            if(n%2 == 0){
                n = n/2;
            }
            else{
                n = n-1;
                ans++;
            }
        }

           ans  = ans + result[n]; 
        }
       

        return ans;
    }
}
