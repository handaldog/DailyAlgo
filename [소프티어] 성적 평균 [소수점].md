import java.io.*;
import java.util.*;

public class Main {

    public static void main(String[] args) {

      Scanner sc = new Scanner(System.in);

      int n = sc.nextInt();
      int k = sc.nextInt();
      

      int nn[] = new int [n+1];

      for(int i=1;i<=n;i++){
        nn[i] = sc.nextInt();
      }

      for(int i=0;i<k;i++){
        int a = sc.nextInt();
        int b = sc.nextInt();

        int sum = 0;
        int cnt =0;
        
        for(int j=a;j<=b;j++){
          sum += nn[j];
          cnt++;
        }
        double res = sum/(cnt*1.0);
       
        System.out.print(String.format("%.2f",res));
        

        System.out.print("\n");
      }
    }
}
