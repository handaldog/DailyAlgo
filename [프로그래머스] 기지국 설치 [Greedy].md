class Solution {
    public int solution(int n, int[] stations, int w) {
        
        int answer = 0;
        
        int size = (w*2)+1;

        int center = 1;
        
        for(int i=0;i<stations.length;i++){
            int min = stations[i] - w;
            
            // ìµœì†Œê°’ > center
            if(min > center){
                int check = ((((stations[i] + w) - size)-center) +1);
                if(check%size == 0){
                answer += check/size;
            }
                else {
                    answer+= (check/size)+1;
                 }
               center = stations[i] +w +1;
            }
            
            // ìµœì†Œê°’ < center
            else if(min <= center)center = stations[i] +w +1;
            
        }
        
        // ì›ë˜ ê¸°ì§€êµ­ì„ ë‹¤ í•´ê²°í–ˆëŠ”ë°ë„ ë‚¨ì•„ìˆëŠ” ì•„íŒŒíŠ¸ê°€ ìˆë‹¤ë©´
        if(center < n){
           
            if((n-center)%size == 0){
                answer += (n-center)/size;
            }
            else {
               answer+= ((n-center)/size)+1;
            }
            
        }
        else if(center == n){
            answer++;
        }
        
        
        return answer;
    }
}
